32      public void processIncomingVehicle() {
33          try {
34              ParkingSpot parkingSpot = getNextParkingNumberIfAvailable();
35              if(parkingSpot !=null && parkingSpot.getId() > 0){
36                  String vehicleRegNumber = getVehicleRegNumber();
37                  parkingSpot.setAvailable(false);
38                  parkingSpotDAO.updateParking(parkingSpot);//allot this parking space and mark it's availability as false
39  
40  
41                  Date inTime = new Date();
42                  Ticket ticket = new Ticket();
43                  //ID, PARKING_NUMBER, VEHICLE_REG_NUMBER, PRICE, IN_TIME, OUT_TIME)
44                  //ticket.setId(ticketID);
45                  ticket.setParkingSpot(parkingSpot);
46                  ticket.setVehicleRegNumber(vehicleRegNumber);
47                  ticket.setPrice(0);
48                  ticket.setInTime(inTime);
49                  ticket.setOutTime(null);
50              //check if the vehicle number has already been registered
51              if (ticket.canApplyDiscount()){
52                  System.out.println("Welcome back! As a recurring user of our parking lot, you'll benefit from a 5% discount.");
53              }
54                  ticketDAO.saveTicket(ticket);
55                  System.out.println("Generated Ticket and saved in DB");
56                  System.out.println("Please park your vehicle in spot number:"+parkingSpot.getId());
57                  System.out.println("Recorded in-time for vehicle number:"+vehicleRegNumber+" is:"+inTime);
58  
59              }
60  
61  
62          }catch(Exception e){
63              logger.error("Unable to process incoming vehicle",e);
64          }
65      }

67      private String getVehicleRegNumber() throws Exception {

74      public ParkingSpot getNextParkingNumberIfAvailable() {
75          int parkingNumber;
76          ParkingSpot parkingSpot = null;
77          try {
78              ParkingType parkingType = getVehichleType();
79              parkingNumber = parkingSpotDAO.getNextAvailableSlot(parkingType);
80              if (parkingNumber > 0) {
81                  parkingSpot = new ParkingSpot(parkingNumber,parkingType, true);
82              } else {
83                  throw new Exception("Error fetching parking number from DB. Parking slots might be full");
84              }
85          } catch(IllegalArgumentException ie) {
86              logger.error("Error parsing user input for type of vehicle", ie);
87          } catch(Exception e) {
88              logger.error("Error fetching next available parking slot", e);
89          }
90          return parkingSpot;
91      }

93      private ParkingType getVehichleType() {
94          System.out.println("Please select vehicle type from menu");
95          System.out.println("1 CAR");
96          System.out.println("2 BIKE");
97          int input = inputReaderUtil.readSelection();
98          switch(input) {
99              case 1: {
100                 return ParkingType.CAR;
101             }
102             case 2: {
103                 return ParkingType.BIKE;
104             }
105             default: {
106                 System.out.println("Incorrect input provided");
107                 throw new IllegalArgumentException("Entered input is invalid");
108             }
109         }

112     public void processExitingVehicle() {
113         try {
114             String vehicleRegNumber = getVehicleRegNumber();
115             Ticket ticket = ticketDAO.getTicket(vehicleRegNumber);
116             Date outTime = new Date();
117 
118             ticket.setOutTime(outTime);
119             fareCalculatorService.calculateFare(ticket);
120 
121             //Apply a 5% discount for recurring users
122             if (ticket.canApplyDiscount()) {
123                  System.out.println(" Apply 5% discount for recurring users : " + ticket.getPrice());
124             }
125 
126             if (ticketDAO.updateTicket(ticket)) {
127                 ParkingSpot parkingSpot = ticket.getParkingSpot();
128                 parkingSpot.setAvailable(true);
129                 parkingSpotDAO.updateParking(parkingSpot);
130                 System.out.println("Please pay the parking fare : " + ticket.getPrice());
131                 System.out.println("Recorded out-time for vehicle number: " + ticket.getVehicleRegNumber() + " is : " + outTime);
132 
133             } else {
134                 System.out.println("Unable to update ticket information. Error occurred");
135             }
136 
137         } catch(Exception e) {
138             logger.error("Unable to process exiting vehicle", e);
139         }
140     }
